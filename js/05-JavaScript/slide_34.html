JavaScript Event Handling Example

 

<input id="clickme" type="button" value="click me"
onmouseover="document.getElementById('msg').innerHTML='Come on! Click me'"
onmouseout="document.getElementById('msg').innerHTML='Hey! come back!'" onmousedown="mouseDown()" />

<div id="msg" style="width: 50%; border-style: solid"> </div>
<script type="text/javascript">
   var counter = 0;
   function mouseDown(){
     counter++;
     document.getElementById('msg').innerHTML="Thank you!";
  }
  function mouseUp(){
     window.alert("You have clicked me "+counter+" times");
  }
  document.getElementById("clickme").onmouseup = mouseUp;
</script>
